```php

```

```python
a = [1, 2, 3]
i, j, k = a
```

```python
a = [0, 1, 2, 3, 4, 5, 6, 7] 
print(a[2:5])# 2以降で5未満
[2, 3, 4]

print(a[4:]) # 4より後
[4, 5, 6, 7]

print(a[:4]) 
[0, 1, 2, 3] # 4より手前
print(a[-1:-4]) # [4, 5, 6] 
[4, 5, 6] 
print(a[::3]) # 等間隔
[0, 3, 6]
print(a[::-1]) # 逆順
[7, 6, 5, 4, 3, 2, 1, 0]
```
